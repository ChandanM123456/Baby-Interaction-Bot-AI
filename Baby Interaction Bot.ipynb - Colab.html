def _text_to_speech_output(text, mood_hint="neutral", lang_code="en"):
    """Generates and plays speech using gTTS or a placeholder."""
    if not text: return
    try:
        tts = gTTS(text=text, lang=lang_code, slow=False)
        output_path = "bot_response.mp3"
        tts.save(output_path)
        audio_data = AudioSegment.from_file(output_path)
        # Indicate bot is speaking to prevent overlap
        bot_speaking_event.set()
        pydub_play(audio_data)
        bot_speaking_event.clear() # Clear after speaking
        os.remove(output_path) # Clean up
    except Exception as e:
        print(f"Error generating/playing TTS (gTTS likely needs internet/has limits): {e}")
        print(f"Bot says (fallback): {text}")
        bot_speaking_event.set() # Still block for a short duration
        time.sleep(len(text.split()) * 0.3) # Simulate speaking time
        bot_speaking_event.clear()
def _play_online_audio(url, content_play_event_obj):
    """Downloads and plays audio from a URL."""
    if not url: return
    print(f"Attempting to stream audio from: {url}")
    try:
        response = requests.get(url, stream=True)
        response.raise_for_status() # Raise HTTPError for bad responses (4xx or 5xx)
        temp_audio_path = "temp_online_audio.mp3"
        with open(temp_audio_path, 'wb') as f:
            for chunk in response.iter_content(chunk_size=8192):
                f.write(chunk)
        audio_data = AudioSegment.from_file(temp_audio_path)
        content_play_event_obj.set() # Signal that content is starting
        pydub_play(audio_data)
        content_play_event_obj.clear() # Signal that content has finished
        os.remove(temp_audio_path) # Clean up
        print("Finished playing online audio.")
    except requests.exceptions.RequestException as e:
        print(f"Error downloading/streaming audio from {url}: {e}")
    except Exception as e:
        print(f"Error playing downloaded audio: {e}")
    finally:
        content_play_event_obj.clear() # Ensure this is reset even on error
# Shared events for managing audio playback
bot_speaking_event = threading.Event() # To prevent bot's TTS from overlapping with itself
content_play_event = threading.Event() # To signal when online content audio finishes
# --- CONVERSATION & DIALOGUE MANAGEMENT ---
class ConversationManager:
    def __init__(self):
        self.current_language = DEFAULT_LANGUAGE
        self.child_language_model = collections.defaultdict(int) # Tracks child's recognized words/sounds
        self.conversation_history = collections.deque(maxlen=20) # Store last 20 interactions
        self.current_context = "general" # e.g., "story", "rhyme", "vocab_practice", "listening_to_request"
        self.active_content_info = None # Stores info about currently playing song/story
        self.last_bot_output_url = None # To prevent immediate repetition of content
        self.last_bot_output_text = None # To prevent immediate repetition of phrases
        self.turn_count_in_context = 0 # How many turns in current specific activity
        self.request_pending = False # Flag if baby made a request and is awaiting content
        # Basic conversational states
        self.states = ["greeting", "listening", "responding_to_mood", "playing_content",
                       "prompting_speech", "asking_question", "farewell"]
        self.current_state = "greeting"
    def _get_random_greeting(self):
        greetings = {
            "English": ["Hello, little one!", "Hi there, baby!", "Good to see you!", "What a bright smile you have!"],
            "Hindi": [",  !", " !", "   !", "    !"]
        }
        return random.choice(greetings[self.current_language])
    def _get_random_encouragement(self):
        encouragements = {
            "English": ["That's a great sound!", "Keep trying, you're doing so well!", "I hear you! What else can you say?", "Wonderful sounds, little one!"],
            "Hindi": ["   !", "  ,      !", "    !     ?", " ,  !"]
        }
        return random.choice(encouragements[self.current_language])
    def _get_random_question(self):
        questions = {
            "English": ["What do you see around you?", "Can you tell me something?", "What's that sound?", "Are you having fun today?", "What makes you happy?"],
            "Hindi": ["  -    ?", "      ?", "    ?", "      ?", "    ?"]
        }
        return random.choice(questions[self.current_language])
    def _select_content_based_on_mood(self, mood, content_type, specific_name=None):
        content_map = {
            "lullaby": LULLABIES_BY_MOOD,
            "rhyme": RHYMES_BY_MOOD,
            "story": STORIES_BY_MOOD
        }
        if content_type not in content_map or not content_map[content_type]:
            # print(f"Warning: No content available for type '{content_type}'")
            return None
        potential_content = []
        # If a specific name is requested
        if specific_name:
            for key_mood, items in content_map[content_type].items():
                for item in items:
                    if item["lang"] == self.current_language and specific_name.lower() in item["name"].lower():
                        potential_content.append(item)
                        break # Found, no need to check other items in this mood
                if potential_content: break # Found, no need to check other moods
        # If no specific name, or specific name not found, try mood-based selection
        if not potential_content:
            # Try to find direct mood-specific content
            for key_mood, items in content_map[content_type].items():
                if mood == key_mood:
                    potential_content.extend([item for item in items if item["lang"] == self.current_language])
            # If no direct mood match, broaden search based on mood type
            if not potential_content:
                if mood in ["calm", "neutral", "sleepy"]:
                    potential_content.extend([item for key_mood, items in content_map[content_type].items()
                                              for item in items if item["lang"] == self.current_language and key_mood in ["sleepy", "engaged", "calm", "neutral"]])
                elif mood in ["happy", "engaged", "fussy"]: # Fussy might also benefit from engagement if not sleeping
                    potential_content.extend([item for key_mood, items in content_map[content_type].items()
                                              for item in items if item["lang"] == self.current_language and key_mood in ["happy", "engaged", "neutral"]])
        # Fallback to any content of the type and language if no specific name or mood match
        if not potential_content:
            for items in content_map[content_type].values():
                 potential_content.extend([item for item in items if item["lang"] == self.current_language])
        if potential_content:
            # Filter out recently played content to avoid immediate repetition
            available_content = [item for item in potential_content if item.get("url") != self.last_bot_output_url]
            if not available_content: # If all are recently played, just pick one from original list to avoid infinite loop
                available_content = potential_content
            selected_item = random.choice(available_content)
            self.last_bot_output_url = selected_item.get("url") # Update last played
            return selected_item
        # print(f"No suitable content found for mood '{mood}', type '{content_type}', specific name '{specific_name}' in language '{self.current_language}'")
        return None
    def _suggest_new_vocabulary_word(self):
        # Suggest a word the child hasn't said much yet
        current_vocab = VOCABULARY.get(self.current_language, [])
        unfamiliar_words = [word for word in current_vocab if self.child_language_model[word.lower()] < 2] # Said < 2 times
        if unfamiliar_words:
            return random.choice(unfamiliar_words)
        return None # All words are familiar or no vocab for this lang
    def _analyze_vocalization_for_keywords(self, vocalization_text):
        if not vocalization_text:
            return {"type": "none"}
        text_lower = vocalization_text.lower()
        # Check for specific content requests first
        request_keywords = {
            "story": ["story", "", "tell story", "read book"],
            "song": ["song", "", "sing", "music"],
            "rhyme": ["rhyme", "", "poem"],
        }
        for req_type, keywords in request_keywords.items():
            for keyword in keywords:
                if keyword in text_lower:
                    self.child_language_model[keyword] += 1
                    return {"type": "request", "content_type": req_type, "keyword": keyword} # Return immediately on first match
        # Check for general vocabulary
        for lang_vocab_list in VOCABULARY.values():
            for word in lang_vocab_list:
                if word.lower() in text_lower:
                    self.child_language_model[word.lower()] += 1
                    return {"type": "vocab", "word": word.lower()}
        return {"type": "none"}
    def process_interaction(self, mood, vocalization_text, content_playing_flag):
        """
        Determines the bot's response based on mood, vocalization, and current state.
        Returns (text_response, audio_url_for_content, new_state).
        """
        self.bot_response_text = ""
        self.bot_response_audio_url = ""
        new_state = self.current_state # Default to no change
        # Store interaction for history and debugging
        self.conversation_history.append({"mood": mood, "vocalization": vocalization_text, "time": time.time(), "state": self.current_state})
        analyzed_vocalization = self._analyze_vocalization_for_keywords(vocalization_text)
        recognized_keyword = analyzed_vocalization.get("word") if analyzed_vocalization.get("type") == "vocab" else None
        content_request = analyzed_vocalization.get("type") == "request"
        # --- Handle Content Playing State ---
        # If content is currently playing, bot remains quiet or gives minimal acknowledgment.
        if content_playing_flag:
            # We assume the _play_online_audio function (in its thread) will manage
            # content_play_event.is_set() which then signals the main loop to handle post-content actions.
            # So, if this flag is True, we just return to keep the bot quiet.
            return "", "", "playing_content"
        # --- Handle Post-Content Transition (executed by main loop once content_play_event clears) ---
        # This logic should execute *only after* content_play_event.is_set() becomes False,
        # indicating content has finished or failed. This is primarily handled by the main loop.
        # But we also need to clear state here if we were in 'playing_content' and it's no longer playing.
        if self.current_state == "playing_content" and not content_playing_flag:
            # Content just finished (or failed) in the previous cycle
            if self.active_content_info and self.active_content_info.get("prompt_after"):
                self.bot_response_text = self.active_content_info["prompt_after"]
                new_state = "asking_question"
            else:
                self.bot_response_text = self._get_random_question() # Or a generic "What next?"
                new_state = "listening"
            self.active_content_info = None # Clear active content info
            self.request_pending = False # Clear any pending request
            self.turn_count_in_context = 0 # Reset turn count
            self.last_bot_output_text = self.bot_response_text # Update last output
            return self.bot_response_text, self.bot_response_audio_url, new_state
        # --- Handle Specific Requests from Baby ---
        if content_request:
            req_type = analyzed_vocalization["content_type"]
            self.request_pending = True
            self.current_context = "listening_to_request" # Set a context for fulfilling request
            self.bot_response_text = "" # Clear previous response text before setting new one
            if req_type == "story":
                selected_content = self._select_content_based_on_mood(mood, "story", specific_name=analyzed_vocalization.get("keyword"))
                if selected_content:
                    self.bot_response_text = f"Oh, you want a story! That's a great idea! Let's listen to '{selected_content['name']}'."
                    self.bot_response_audio_url = selected_content["url"]
                    self.active_content_info = selected_content
                    self.current_context = "story"
                    new_state = "playing_content"
                else:
                    self.bot_response_text = "I'm sorry, little one, I can't find that story right now. How about we sing a happy rhyme instead?"
                    new_state = "responding_to_mood"
                    self.request_pending = False # Request failed, try something else
            elif req_type == "song" or req_type == "rhyme":
                # Prioritize a specific name if available in request, otherwise mood-based
                specific_name_from_vocalization = analyzed_vocalization.get("keyword")
                # Filter out generic request keywords like "song", "rhyme" from being treated as specific names
                if specific_name_from_vocalization in ["song", "rhyme", "", "", "sing"]:
                    specific_name_from_vocalization = None
                selected_content = self._select_content_based_on_mood(
                    mood,
                    "rhyme" if req_type == "rhyme" else "happy", # Default to rhyme for song requests or if specific name not found
                    specific_name=specific_name_from_vocalization
                )
                if not selected_content: # If specific search fails, try mood-based broadly
                     selected_content = self._select_content_based_on_mood(mood, "rhyme") # Fallback to any rhyme for mood
                     if not selected_content: # Absolute fallback to any happy rhyme
                         selected_content = self._select_content_based_on_mood("happy", "rhyme")
                if selected_content:
                    self.bot_response_text = f"You want a song! Let's sing '{selected_content['name']}' together!"
                    self.bot_response_audio_url = selected_content["url"]
                    self.active_content_info = selected_content
                    self.current_context = "rhyme" # Using rhyme context for both songs/rhymes for simplicity
                    new_state = "playing_content"
                else:
                    self.bot_response_text = "I'm sorry, little one, I can't find that song or rhyme right now. How about we make some fun sounds?"
                    new_state = "responding_to_mood"
                    self.request_pending = False # Request failed, try something else
            self.last_bot_output_text = self.bot_response_text # Update last output
            return self.bot_response_text, self.bot_response_audio_url, new_state
        # --- Main Mood-Based Logic (if no specific request and not actively playing content) ---
        # Only proceed if we're not currently fulfilling a request or already transitioning from content
        if not self.request_pending and self.current_state not in ["playing_content", "asking_question"]:
            if mood == "distressed":
                selected_lullaby = self._select_content_based_on_mood("sleepy", "lullaby")
                if selected_lullaby:
                    self.bot_response_text = "Oh, my dear. It looks like you're feeling sad. It's okay to cry. Let me play a very gentle lullaby for you."
                    self.bot_response_audio_url = selected_lullaby["url"]
                    self.active_content_info = selected_lullaby
                    self.current_context = "lullaby"
                    new_state = "playing_content"
                else:
                    self.bot_response_text = "It looks like you're feeling sad. I'm here for you, little one. I hope you feel better soon."
                    new_state = "responding_to_mood"
            elif mood == "fussy":
                selected_story = self._select_content_based_on_mood("calm", "story")
                if selected_story:
                    self.bot_response_text = "Are you feeling a little restless, little one? Perhaps a calm story will help you relax."
                    self.bot_response_audio_url = selected_story["url"]
                    self.active_content_info = selected_story
                    self.current_context = "story"
                    new_state = "playing_content"
                else:
                    self.bot_response_text = "It looks like you're feeling restless. Let's try some calm, soft sounds."
                    selected_lullaby = self._select_content_based_on_mood("fussy", "lullaby") # Try fussy lullaby
                    if selected_lullaby:
                        self.bot_response_audio_url = selected_lullaby["url"]
                        self.active_content_info = selected_lullaby
                        self.current_context = "lullaby"
                        new_state = "playing_content"
                    else:
                        self.bot_response_text += " I hope you feel better soon."
                    new_state = "responding_to_mood"
            elif mood == "happy":
                if recognized_keyword:
                    self.bot_response_text = f"Yes, you said '{recognized_keyword}'! That's wonderful!"
                    if recognized_keyword in ["hello", "hi", ""]:
                        self.bot_response_text = self._get_random_greeting() + " You said hello!"
                    elif recognized_keyword in ["ball", ""]:
                        self.bot_response_text = f"Yes, a {recognized_keyword}! That's a fun toy! Can you roll the {recognized_keyword}?"
                    elif recognized_keyword in ["mommy", "mummy", ""]:
                        self.bot_response_text = "You said Mommy! She loves you very much!"
                    new_state = "prompting_speech"
                elif vocalization_text:
                    self.bot_response_text = self._get_random_encouragement() + f" What else can you say?"
                    new_state = "prompting_speech"
                else:
                    selected_rhyme = self._select_content_based_on_mood("happy", "rhyme")
                    if selected_rhyme:
                        self.bot_response_text = "You look so happy! Let's sing a fun rhyme together!"
                        self.bot_response_audio_url = selected_rhyme["url"]
                        self.active_content_info = selected_rhyme
                        self.current_context = "rhyme"
                        new_state = "playing_content"
                    else:
                        self.bot_response_text = self._get_random_greeting() + " What a lovely smile!"
                    new_state = "responding_to_mood"
            elif mood == "sleepy":
                selected_lullaby = self._select_content_based_on_mood("sleepy", "lullaby")
                if selected_lullaby:
                    self.bot_response_text = "Are your eyes feeling heavy, little one? It's time for a sweet lullaby."
                    self.bot_response_audio_url = selected_lullaby["url"]
                    self.active_content_info = selected_lullaby
                    self.current_context = "lullaby"
                    new_state = "playing_content"
                else:
                    self.bot_response_text = "It looks like you're sleepy. Let's listen to some gentle, calming sounds."
                    new_state = "responding_to_mood"
            elif mood == "neutral" or mood == "engaged":
                if recognized_keyword:
                    self.bot_response_text = f"You said '{recognized_keyword}'! Can you say it again?"
                    new_state = "prompting_speech"
                elif vocalization_text:
                    self.bot_response_text = self._get_random_encouragement() + " Keep making those wonderful sounds!"
                    new_state = "prompting_speech"
                else:
                    if self.current_state == "greeting":
                        self.bot_response_text = self._get_random_greeting() + " What a lovely day!"
                        new_state = "listening"
                    elif self.turn_count_in_context % 5 == 0:
                        self.bot_response_text = self._get_random_question()
                        new_state = "asking_question"
                    else:
                        suggested_word = self._suggest_new_vocabulary_word()
                        if suggested_word:
                            self.bot_response_text = f"Can you say '{suggested_word}'?"
                            new_state = "prompting_speech"
                        else:
                            selected_story = self._select_content_based_on_mood("engaged", "story")
                            if selected_story:
                                self.bot_response_text = "Let's read a short story!"
                                self.bot_response_audio_url = selected_story["url"]
                                self.active_content_info = selected_story
                                self.current_context = "story"
                                new_state = "playing_content"
                            else:
                                self.bot_response_text = "How about we explore some new sounds together?"
                            new_state = "responding_to_mood"
        # If no specific action was taken and no text was set, provide a default engaging response
        if not self.bot_response_text:
            if mood in ["happy", "neutral", "engaged"]:
                self.bot_response_text = random.choice([self._get_random_question(), "Tell me more, little one!", "Let's discover something new!"])
            elif mood == "distressed":
                self.bot_response_text = "I'm here for you, sweetie. It's okay."
            elif mood == "sleepy":
                self.bot_response_text = "Just rest, little one."
            elif mood == "fussy":
                self.bot_response_text = "Let's try to find something calming for you."
        # Prevent immediate repetition of text responses if no new input leads to a new response
        if self.last_bot_output_text and self.bot_response_text == self.last_bot_output_text and not (vocalization_text or content_request):
            if mood in ["happy", "neutral", "engaged"]:
                self.bot_response_text = random.choice([self._get_random_question(), "What's on your mind?", "Let's imagine something fun!"])
            else:
                self.bot_response_text = "I'm right here for you."
        self.last_bot_output_text = self.bot_response_text
        return self.bot_response_text, self.bot_response_audio_url, new_state
    def update_language(self, lang_code):
        if lang_code in VOCABULARY:
            self.current_language = lang_code
            print(f"Bot language set to: {self.current_language}")
            # Greet in new language and trigger a response
            self.bot_response_text = self._get_random_greeting()
            return True
        return False

class BabyBotColab:
    def __init__(self):
        self.conversation_manager = ConversationManager()

        self.is_content_playing = False # Flag for main loop
        self.mood_detection_interval = 2 # seconds, how often to check mood
        self.last_mood_check_time = time.time()
        if DEEPFACE_AVAILABLE:
            print("Using DeepFace for emotion detection.")
        else:
            print("Using basic OpenCV face detection (no advanced mood recognition without DeepFace).")
    def _process_frame_for_mood(self, frame):
        """Analyzes a single frame for mood."""
        try:
            if DEEPFACE_AVAILABLE:
                # DeepFace returns a list of dictionaries for each face
                # enforce_detection=False allows it to proceed even if detection confidence is low
                detections = DeepFace.analyze(frame, actions=['emotion'], enforce_detection=False, silent=True)
                if detections and len(detections) > 0:
                    dominant_emotion = detections[0]["dominant_emotion"]
                    # Map DeepFace emotions to simpler bot moods
                    if dominant_emotion in ["happy", "surprise"]: return "happy"
                    if dominant_emotion in ["sad", "fear", "angry", "disgust"]: return "distressed"
                    if dominant_emotion == "neutral": return "neutral"
                return "neutral" # Default if no face or no clear emotion
            else:
                # Fallback: simple face detection, no mood analysis
                gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
                face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_frontalface_default.xml')
                faces = face_cascade.detectMultiScale(gray, 1.3, 5)
                if len(faces) > 0:
                    return "engaged" # If a face is detected, assume engaged/neutral
                return "neutral" # No face detected
        except Exception as e:
            # print(f"Error in mood analysis: {e}. Returning neutral.") # Uncomment for debugging mood analysis issues
            return "neutral"
    def _recognize_speech_from_audio_buffer(self, audio_bytes):
        """
        Processes audio data for speech recognition (simulated keyword spotting).
        """
        if not audio_bytes or len(audio_bytes) < 1000: # Ignore very small or empty chunks
            return ""
        simulated_baby_vocalizations = [
            "goo goo", "ba ba", "ma ma", "da da", "apple", "ball", "dog", "cat", "no", "yes",
            "story", "song", "rhyme", "mama", "dada", "hello", "bye", "milk", "water",
            "", "", "", "", "", "", "", ""
        ]
        # Simulate an occasional, random accurate recognition if audio is present
        # This makes the "baby ask" feature more likely to be demonstrated.
        if random.random() < 0.6: # 60% chance of "recognizing" something from a pre-defined list
            return random.choice(simulated_baby_vocalizations)
        return "" # No clear speech recognized in this simulated pass
    def run_bot(self):
        print("Starting Baby Bot (Colab Simulation). Press Ctrl+C to stop.")
        print("Ensure you have granted camera and microphone permissions.")
        print("Note: Real-time performance will vary greatly due to Colab environment and network.")
        print("\n" + "="*50 + "\n")
        # Initial greeting from bot
        initial_greeting_text = self.conversation_manager._get_random_greeting()
        # Use a separate thread for initial greeting so main loop can start
        threading.Thread(target=_text_to_speech_output, args=(initial_greeting_text, "happy", self.conversation_manager.current_language.lower()[:2])).start()
        # Give some time for initial greeting to play
        time.sleep(3)
        try:
            # Setup camera and audio capture (Colab specific JS)
            display(Javascript(JAVASCRIPT_CAMERA_CODE))
            display(Javascript(JAVASCRIPT_AUDIO_CODE))
            print("JavaScript camera and audio setup initiated. Grant permissions if prompted.")
            frame_count = 0
            # Allow some time for JS to initialize camera/mic and get first frames
            time.sleep(2)
            while True:
                current_time = time.time()
                current_mood = "neutral" # Default if not updated
                # 1. Get Camera Frame and Analyze Mood (less frequent for performance)
                if current_time - self.last_mood_check_time >= self.mood_detection_interval:
                    try:
                        frame_string = eval_js('takePhoto();')
                        if frame_string and frame_string != 'NO_FRAME' and frame_string != 'ERROR_PHOTO': # Check for valid frame
                            jpeg_bytes = b64decode(frame_string.split(',')[1])
                            img_array = np.frombuffer(jpeg_bytes, dtype=np.uint8)
                            frame = cv2.imdecode(img_array, cv2.IMREAD_COLOR)
                            current_mood = self._process_frame_for_mood(frame)
                            _display_frame(frame, current_mood) # Display the frame with mood
                        else:
                            # print(f"Warning: No camera frame captured or error: {frame_string}") # Uncomment for debugging
                            pass # Suppress frequent warnings for no frame
                    except Exception as e:
                        print(f"Error during camera capture or mood analysis: {e}")
                        current_mood = "neutral" # Fallback
                    self.last_mood_check_time = current_time # Reset timer
                # 2. Get Audio Input and Recognize Speech (simulated)
                audio_input_base64 = "" # Reset for current cycle
                try:
                    audio_input_base64 = eval_js('getAudioData();')
                except Exception as e:
                    print(f"Error getting audio data from JS: {e}")
                baby_vocalization = ""
                if audio_input_base64 and audio_input_base64 != 'NO_AUDIO' and audio_input_base64 != 'ERROR_AUDIO':
                    try:
                        # Extract base64 data correctly
                        if ',' in audio_input_base64:
                            audio_data_raw = audio_input_base64.split(',')[1]
                        else:
                            audio_data_raw = audio_input_base64 # Might already be just base64
                        audio_bytes = b64decode(audio_data_raw)
                        baby_vocalization = self._recognize_speech_from_audio_buffer(audio_bytes)
                        if baby_vocalization:
                            print(f"Baby Vocalization (Simulated ASR): '{baby_vocalization}'")
                    except Exception as e:
                        print(f"Error processing audio data: {e}")
                # 3. Dialogue Management and Response
                # Only proceed if bot is not currently speaking or playing content
                if not bot_speaking_event.is_set() and not content_play_event.is_set():
                    # Check if content just finished playing (flag was True, now event is clear)
                    if self.is_content_playing and not content_play_event.is_set():
                        self.is_content_playing = False # Content finished
                        print("Content playback finished.")
                        # This triggers the post-content dialogue from process_interaction's initial check
                        bot_text, bot_audio_url, new_state = self.conversation_manager.process_interaction(
                            mood=current_mood,
                            vocalization_text="", # No new vocalization during this transition
                            content_playing_flag=False
                        )
                    else: # Normal interaction loop
                        self.is_content_playing = False # Ensure this is false if no content is playing
                        bot_text, bot_audio_url, new_state = self.conversation_manager.process_interaction(
                            mood=current_mood,
                            vocalization_text=baby_vocalization,
                            content_playing_flag=self.is_content_playing
                        )
                    self.conversation_manager.current_state = new_state # Update state
                    # Decide between playing audio content or speaking text
                    if bot_audio_url:
                        print(f"Bot will play content: {bot_audio_url}")
                        # Play in a separate thread to avoid blocking main loop
                        threading.Thread(target=_play_online_audio, args=(bot_audio_url, content_play_event)).start()
                        self.is_content_playing = True # Set flag immediately
                    elif bot_text and not bot_speaking_event.is_set(): # Check again before speaking
                        print(f"Bot will speak: {bot_text}")
                        # Speak in a separate thread
                        threading.Thread(target=_text_to_speech_output, args=(bot_text, current_mood, self.conversation_manager.current_language.lower()[:2])).start()
                # Control loop speed for simulation
                time.sleep(1) # Adjust this to control how often it processes frames/audio
                frame_count += 1
                if frame_count % 5 == 0: # Print status less frequently
                    print(f"\n--- Current Bot Status (Cycle {frame_count}) ---")
                    print(f"Detected Mood: {current_mood}")
                    print(f"Bot Internal State: {self.conversation_manager.current_state}")
                    print(f"Content Playing (Flag): {self.is_content_playing}")
                    print(f"Content Playing (Event): {content_play_event.is_set()}")
                    print(f"Bot Speaking (Event): {bot_speaking_event.is_set()}")
                    print(f"Child Vocab: {dict(self.conversation_manager.child_language_model)}")
                    print("---" * 20)
        except KeyboardInterrupt:
            print("\nStopping bot session...")
        except Exception as e:
            print(f"An unexpected error occurred: {e}")
        finally:

            # Stop JS recording explicitly if possible
            try:
                eval_js('stopRecording();')
            except:
                pass # JS might already be down or permissions revoked
            print("Baby Bot session ended.")
